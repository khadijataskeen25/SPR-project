#include <stdio.h>
#include<stdlib.h>
#include<time.h>
int rps(char user,char computer)
{
    if(user==computer)
    {
        return 0;
    }
    if(user=='R'&&computer=='P')
    {
        return -1;
    }
    else if(user=='R'&&computer=='S')
    {
        return 1;
    }
    if(user=='P'&&computer=='S')
    {
        return -1;
    }
    else if(user=='P'&&computer=='R')
    {
        return 1;
    }
        if(user=='S'&&computer=='R')
    {
        return -1;
    }
    else if(user=='S'&&computer=='P')
    {
        return 1;
    }
}
int main()
{
    char user , computer;
    int player;
    int bot;
    int draw;
    printf("WELCOME TO THE GAME!\n");
    printf("R=Rock!\nS=Scissors\nP=Paper\n");
    
    srand(time(0));
    int number=rand()%100;
    
    if(number<36)
    {
        computer='R';
    }
    else if(number>36&&number<66)
    {
        computer='S';
    }
    else
    {
        computer='P';
    }
    for(int i=0;i<5;i++)
    {
    printf("Enter your turn:\n");
    scanf("%c ",&user);
    
    printf("Computer's turn :%c\n",computer);
    int result=rps(user,computer);
    
    if(result==1)
    {
        printf("You get a point\n");
        player++;
    }
    else if(result==-1)
    {
        printf("Computer gets a point\n");
        bot++;
    }
    else if(result==0)
    {
        printf("ops!its a draw.\n");
        draw++;
    }
    }
    printf("your total score is:%d\n",player);
    printf("computer's total score is :%d\n",bot);
    printf("ops! The match is a draw .\n the score is :%d\n",draw);
    if(player>bot)
    {
        printf("CONGRATULATIONS! you have won the game! ");
    }
    else if(bot>player)
    {
        printf("Computer has won the game.");
    }
    else
    {
        printf("Its a draw.");
    }
    
    return 0;
}
